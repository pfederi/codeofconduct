// Firebase Storage Security Rules
// Kopiere diese Rules in die Firebase Console unter Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Supporters folder - allow read for everyone, write for everyone (for initial upload)
    // After upload, you can restrict write access to authenticated users only
    match /supporters/{allPaths=**} {
      allow read: if true;  // Jeder kann lesen
      allow write: if true; // Temporär: Jeder kann schreiben (für Upload)
      // Nach dem Upload ändern zu:
      // allow write: if request.auth != null;  // Nur authentifizierte Benutzer können schreiben
    }
    
    // Default: deny all other access
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
