// Firebase Firestore Security Rules
// Kopiere diese Rules in die Firebase Console unter Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Supporters collection - allow read for everyone, write for everyone (for initial setup)
    match /supporters/{supporterId} {
      allow read: if true;  // Jeder kann lesen
      allow write: if true; // Temporär: Jeder kann schreiben (für Upload)
      // Nach dem Upload ändern zu:
      // allow write: if request.auth != null;  // Nur authentifizierte Benutzer können schreiben
    }
    
    // Signatories collection - existing rules
    match /signatories/{signatoryId} {
      allow read: if true;
      allow create: if true;  // Jeder kann neue Unterschriften erstellen
      allow update, delete: if false;  // Nur Lesen und Erstellen erlaubt
    }
  }
}
